    71→function LoginForm({ onSwitchToRegister, onLoginSuccess }: { onSwitchToRegister: () => void; onLoginSuccess: (user: any, token: string) => void }) {
    72→  const [email, setEmail] = useState('');
    73→  const [password, setPassword] = useState('');
    74→  const [showPassword, setShowPassword] = useState(false);
    75→  const [error, setError] = useState('');
    76→  const [loading, setLoading] = useState(false);
    77→  const [focused, setFocused] = useState<string | null>(null);
    78→
    79→  async function handleSubmit(e: React.FormEvent) {
    80→    e.preventDefault();
    81→    setError('');
    82→    if (!email.trim()) { setError('Email is required'); return; }
    83→    if (!password) { setError('Password is required'); return; }
    84→    setLoading(true);
    85→    try {
    86→      const res = await fetch('/api/auth/login', {
    87→        method: 'POST',
    88→        headers: { 'Content-Type': 'application/json' },
    89→        body: JSON.stringify({ email: email.trim(), password }),
    90→      });
    91→      const data = await res.json();
    92→      if (!res.ok) { setError(data.error || 'Login failed. Please try again.'); return; }
    93→      onLoginSuccess(data.user, data.token);
    94→    } catch { setError('Network error. Please check your connection.'); } finally { setLoading(false); }
    95→  }
    96→
    97→  return (
    98→    <form onSubmit={handleSubmit} className="flex flex-col gap-5">
    99→      <div>
   100→        <label className="block mb-2" style={{ fontSize: 12, fontWeight: 500, color: '#A1A1AA', letterSpacing: '0.04em', textTransform: 'uppercase' }}>Email Address</label>

   106→        </motion.div>

   108→      <div>
   109→        <label className="block mb-2" style={{ fontSize: 12, fontWeight: 500, color: '#A1A1AA', letterSpacing: '0.04em', textTransform: 'uppercase' }}>Password</label>

   114→            <button type="button" className="absolute right-3.5 top-1/2 -translate-y-1/2 bg-transparent border-none cursor-pointer" style={{ color: '#71717A' }} onClick={() => setShowPassword(!showPassword)}>

   144→/* ─── REGISTER FORM ─── */
   145→function RegisterForm({ onSwitchToLogin, onRegisterSuccess }: { onSwitchToLogin: () => void; onRegisterSuccess: (user: any, token: string) => void }) {
   146→  const [name, setName] = useState('');
   147→  const [email, setEmail] = useState('');
   148→  const [phone, setPhone] = useState('');
   149→  const [password, setPassword] = useState('');
   150→  const [confirmPassword, setConfirmPassword] = useState('');
   151→  const [code, setCode] = useState('');
   152→  const [showPassword, setShowPassword] = useState(false);
   153→  const [showConfirm, setShowConfirm] = useState(false);
   154→  const [error, setError] = useState('');
   155→  const [errors, setErrors] = useState<Record<string, string>>({});
   156→  const [loading, setLoading] = useState(false);
   157→
   158→  useEffect(() => {
   159→    const saved = localStorage.getItem('nextrade_invitation_code');
   160→    if (saved) { setCode(saved); localStorage.removeItem('nextrade_invitation_code'); }
   161→  }, []);
   162→
   163→  function validate(): boolean {
   164→    const errs: Record<string, string> = {};
   165→    if (!name.trim()) errs.name = 'Name is required';
   166→    else if (name.trim().length < 2) errs.name = 'Name must be at least 2 characters';
   167→    if (!email.trim()) errs.email = 'Email is required';
   168→    else if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email.trim())) errs.email = 'Invalid email format';
   169→    if (!password) errs.password = 'Password is required';
   170→    else if (password.length < 6) errs.password = 'Password must be at least 6 characters';
   171→    if (!confirmPassword) errs.confirmPassword = 'Please confirm your password';
   172→    else if (password !== confirmPassword) errs.confirmPassword = 'Passwords do not match';
   173→    setErrors(errs);
   174→    return Object.keys(errs).length === 0;
   175→  }
   176→
   177→  async function handleSubmit(e: React.FormEvent) {
   178→    e.preventDefault();
   179→    setError('');
   180→    if (!validate()) return;
   181→    setLoading(true);
   182→    try {
   183→      const res = await fetch('/api/auth/register', {
   184→        method: 'POST',
   185→        headers: { 'Content-Type': 'application/json' },
   186→        body: JSON.stringify({ name: name.trim(), email: email.trim(), phone: phone.trim() || undefined, password, invitationCode: code.trim() || undefined }),
   187→      });
   188→      const data = await res.json();
   189→      if (!res.ok) { setError(data.error || 'Registration failed.'); return; }
   190→      if (data.token && data.user) { onRegisterSuccess(data.user, data.token); }
   191→    } catch { setError('Network error.'); } finally { setLoading(false); }
   192→  }
   193→
   194→  function fieldError(field: string) {
   195→    return errors[field] ? <span style={{ color: '#f87171', fontSize: 12, marginTop: 4, display: 'block' }}>{errors[field]}</span> : null;
   196→  }
   197→
   198→  return (
   199→    <form onSubmit={handleSubmit} className="flex flex-col gap-3.5">
   200→      <div>
   201→        <label className="block mb-1.5" style={{ fontSize: 12, fontWeight: 500, color: '#A1A1AA', textTransform: 'uppercase', letterSpacing: '0.04em' }}>Full Name</label>
   202→        <div className="relative">
   203→          <User size={16} className="absolute left-3.5 top-1/2 -translate-y-1/2" style={{ color: '#71717A' }} />
   204→          <input type="text" className="input-field" style={{ paddingLeft: 42 }} placeholder="John Doe" value={name} onChange={(e) => setName(e.target.value)} autoComplete="name" />
   205→        </div>
   206→        {fieldError('name')}
   207→      </div>
   208→      <div>
   209→        <label className="block mb-1.5" style={{ fontSize: 12, fontWeight: 500, color: '#A1A1AA', textTransform: 'uppercase', letterSpacing: '0.04em' }}>Email Address</label>
   210→        <div className="relative">
   211→          <Mail size={16} className="absolute left-3.5 top-1/2 -translate-y-1/2" style={{ color: '#71717A' }} />
   212→          <input type="email" className="input-field" style={{ paddingLeft: 42 }} placeholder="you@example.com" value={email} onChange={(e) => setEmail(e.target.value)} autoComplete="email" />
   213→        </div>
   214→        {fieldError('email')}
   215→      </div>
   216→      <div>
   217→        <label className="block mb-1.5" style={{ fontSize: 12, fontWeight: 500, color: '#A1A1AA', textTransform: 'uppercase', letterSpacing: '0.04em' }}>Phone Number <span style={{ fontWeight: 400, textTransform: 'none', letterSpacing: 'normal' }}>(optional)</span></label>
   218→        <div className="relative">
   219→          <Phone size={16} className="absolute left-3.5 top-1/2 -translate-y-1/2" style={{ color: '#71717A' }} />
   220→          <input type="tel" className="input-field" style={{ paddingLeft: 42 }} placeholder="+1 (XXX) XXX-XXXX" value={phone} onChange={(e) => setPhone(e.target.value)} autoComplete="tel" />
   221→        </div>
   222→      </div>
   223→      <div>
   224→        <label className="block mb-1.5" style={{ fontSize: 12, fontWeight: 500, color: '#A1A1AA', textTransform: 'uppercase', letterSpacing: '0.04em' }}>Password</label>
   225→        <div className="relative">
   226→          <Lock size={16} className="absolute left-3.5 top-1/2 -translate-y-1/2" style={{ color: '#71717A' }} />
   227→          <input type={showPassword ? 'text' : 'password'} className="input-field" style={{ paddingLeft: 42, paddingRight: 42 }} placeholder="Min 6 characters" value={password} onChange={(e) => setPassword(e.target.value)} autoComplete="new-password" />
   228→          <button type="button" className="absolute right-3.5 top-1/2 -translate-y-1/2 bg-transparent border-none cursor-pointer" style={{ color: '#71717A' }} onClick={() => setShowPassword(!showPassword)}>{showPassword ? <EyeOff size={16} /> : <Eye size={16} />}</button>
   229→        </div>
   230→        {fieldError('password')}
   231→      </div>
   232→      <div>
   233→        <label className="block mb-1.5" style={{ fontSize: 12, fontWeight: 500, color: '#A1A1AA', textTransform: 'uppercase', letterSpacing: '0.04em' }}>Confirm Password</label>
   234→        <div className="relative">
   235→          <Lock size={16} className="absolute left-3.5 top-1/2 -translate-y-1/2" style={{ color: '#71717A' }} />
   236→          <input type={showConfirm ? 'text' : 'password'} className="input-field" style={{ paddingLeft: 42, paddingRight: 42 }} placeholder="Re-enter password" value={confirmPassword} onChange={(e) => setConfirmPassword(e.target.value)} autoComplete="new-password" />
   237→          <button type="button" className="absolute right-3.5 top-1/2 -translate-y-1/2 bg-transparent border-none cursor-pointer" style={{ color: '#71717A' }} onClick={() => setShowConfirm(!showConfirm)}>{showConfirm ? <EyeOff size={16} /> : <Eye size={16} />}</button>
   238→        </div>
   239→        {fieldError('confirmPassword')}
   240→      </div>
   241→      <div>
   242→        <label className="block mb-1.5" style={{ fontSize: 12, fontWeight: 500, color: '#A1A1AA', textTransform: 'uppercase', letterSpacing: '0.04em' }}>Invitation Code <span style={{ fontWeight: 400, textTransform: 'none', letterSpacing: 'normal' }}>(optional)</span></label>
   243→        <div className="relative">
   244→          <Ticket size={16} className="absolute left-3.5 top-1/2 -translate-y-1/2" style={{ color: '#71717A' }} />
   245→          <input type="text" className="input-field" style={{ paddingLeft: 42, textTransform: 'uppercase' }} placeholder="Have a code? Enter here" value={code} onChange={(e) => setCode(e.target.value)} autoComplete="off" />
   246→        </div>
   247→      </div>
   248→      {error && (
   249→        <motion.div initial={{ opacity: 0, y: -5 }} animate={{ opacity: 1, y: 0 }} className="px-3 py-2.5 rounded-lg text-center" style={{ background: 'rgba(239,68,68,0.06)', border: '1px solid rgba(239,68,68,0.15)', color: '#f87171', fontSize: 13 }}>{error}</motion.div>
   250→      )}
   251→      <motion.button type="submit" className="w-full flex items-center justify-center gap-2" style={{ padding: '14px 20px', fontSize: 15, fontWeight: 600, color: '#FAFAFA', border: 'none', borderRadius: 8, cursor: 'pointer', background: loading ? 'var(--bg-hover)' : 'var(--gradient-brand)', boxShadow: loading ? 'none' : 'var(--shadow-sm)', fontFamily: "var(--font-poppins), 'Poppins', sans-serif", marginTop: 4 }} disabled={loading}>
   252→        {loading && <Loader2 size={18} className="animate-spin" />}
   253→        {loading ? 'Creating Account...' : 'Create Account'}
   254→      </motion.button>
   255→      <p className="text-center mt-1" style={{ fontSize: 13, color: '#A1A1AA' }}>
   256→        Already have an account?{' '}
   257→        <button type="button" className="bg-transparent border-none cursor-pointer" style={{ color: '#3B82F6', fontWeight: 600, fontSize: 13 }} onClick={onSwitchToLogin}>Sign In</button>
   258→      </p>
   259→    </form>
   260→  );
   261→}
